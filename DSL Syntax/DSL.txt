Story EnchantedForest
StartNode start {     
    displayText: "You are in an enchanted forest. Magic fills the air and you feel enchanted\n\nYou see 2 roads to go down and some bushes around",
    -> roadChoice
}

ChoiceNode roadChoice {
    choices: [
        ChoiceOption {displayText: "Go down road number 1", 
            -> road1DialogueTalisman with priority 2 with condition: Talisman=1,
            -> road1DialogueNoTalisman},
        ChoiceOption {displayText: "Go down road number 2", 
            -> road2Dialogue},
        ChoiceOption {displayText: "Look in bushes", 
            -> foundTalismanDialogue with priority 2 with condition: Talisman=1,
            -> foundNothingDialogue}
    ]
}

DialogueNode foundTalismanDialogue {
    displayText: "You found a Talisman! It looks important", 
    -> addTalismanSystemStateChange
}

SystemStateChangeNode addTalismanSystemStateChange {
    displayText: "Adds talisman to backpack",
    "Talisman"+=1,
    -> roadChoice
}

DialogueNode foundNothingDialogue {
    displayText: "You find nothing", 
    -> roadChoice
}

DialogueNode road2Dialogue {
    displayText: "You find an oppening in the forrest but wait! It is blocked by a magic Shield!", 
    -> shieldChoice
}

ChoiceNode shieldChoice {
    choices: [
        ChoiceOption {displayText: "Go back to start", 
            -> Start},
        ChoiceOption {displayText: "Try to go through",
            -> goThroughOpeningDialogueSuccess with priority 2 with condition: Talisman=1,
            -> goThroughOpeningDialogueFail}
    ]
}

DialogueNode goThroughOpeningDialogueSuccess {
    displayText: "You get through to the oppening", 
    -> moreChoices...
}

DialogueNode goThroughOpeningDialogueFail {
    displayText: "On no you are blocked!", 
    -> shieldChoice
}

DialogueNode road1DialogueNoTalisman {
    displayText: "You encounter a suspisious looking old lady!\n\"Hello dear child\"\n\n\"Do you perhaps want to help an old lady through the forest?", 
    -> movePastWitchChoice
}

DialogueNode road1DialogueTalisman {
    displayText: "You encounter a suspisious looking old lady!\n\"Hello dear child\"\n\n\"You see the witch for who she is", 
    -> movePastWitchChoice
}

ChoiceNode movePastWitchChoice {
    choices: [
        ChoiceOption {displayText: "Go back to start", 
            -> Start},
        ChoiceOption {displayText: "Go past her",
             -> walkPastWitch with priority 2 with condition: Talisman=1,
             -> witchBadEnd}
    ]
}

DialogueNode walkPastWitch {
    displayText: "You catch a lurking eye from the old lady as you walk past", 
    -> moreChoices...
}

EndNode witchBadEnd {
    displayText: "The witch stabs you, mauls you and absolutely decimates you. You are dead, dead, dead. Any questions?" 
}

EndNode witchBadEnd {
    displayText: "Opps... The story ended abruptly.. To be continued maybe?..." 
}
